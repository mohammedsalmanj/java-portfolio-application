stage("Checkstyle") {
    steps {
        sh "mvn checkstyle:checkstyle"
        publishHTML(target: [
            allowMissing: true,
            alwaysLinkToLastBuild: true,
            keepAll: true,
            reportDir: 'target/site',
            reportFiles: 'checkstyle.html',
            reportName: 'Checkstyle Report'
        ])
    }
}
